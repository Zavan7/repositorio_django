import smtplib
from email.message import EmailMessage
from config import (
    SMTP_HOST,
    SMTP_PORT,
    EMAIL_USER,
    EMAIL_PASS,
    EMAIL_SUBJECT
)


class EmailService:

    def __init__(self):

        self.server = None

        self._connect()

    def _connect(self):

        try:

            self.server = smtplib.SMTP(
                SMTP_HOST,
                SMTP_PORT
            )

            self.server.starttls()

            self.server.login(
                EMAIL_USER,
                EMAIL_PASS
            )

        except Exception as e:

            raise RuntimeError(
                f"Erro ao conectar SMTP: {e}"
            )

    def send(self, to: str, message: str):

        if not to:
            return

        msg = EmailMessage()

        msg["From"] = EMAIL_USER

        msg["To"] = to

        msg["Subject"] = EMAIL_SUBJECT

        msg.set_content(message or "")

        self.server.send_message(msg)

    def close(self):

        if self.server:

            try:

                self.server.quit()

            except Exception:
                pass
